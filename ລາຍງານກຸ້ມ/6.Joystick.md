#### code
```c++
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27, 16, 2);
int X = A0;
int Y = A1;
int SW = A2;
int playerX = 0;
int playerY = 0;
void setup() {
  lcd.init();
  lcd.backlight();
  pinMode(SW, INPUT);
  Serial.begin(9600);
}
void loop() {
  int xVal = analogRead(X);
  int yVal = analogRead(Y);
  // เคลื่อนซ้าย
  if (xVal < 400 && playerX > 0) {
    playerX--;
    delay(150);
  }
  // เคลื่อนขวา
  else if (xVal > 600 && playerX < 15) {
    playerX++;
    delay(150);
  }
  // ขึ้น
  if (yVal < 400 && playerY > 0) {
    playerY--;
    delay(150);
  }
  // ลง
  else if (yVal > 600 && playerY < 1) {
    playerY++;
    delay(150);
  }
  // แสดงผลตำแหน่งตัวละคร
  lcd.clear();
  lcd.setCursor(playerX, playerY);
  lcd.print("@");
  delay(100);
}
```
